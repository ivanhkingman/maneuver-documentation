101  {
102  Memory::clear(m_ltr);

104  LoiterTypes type = getType((IMC::Loiter::LoiterTypeEnum)maneuver->type);
105 
106  switch (type)
107  {
108  case LTYPE_FIGURE_EIGHT:
109  Maneuvers::FigureEight* fig;
110  fig = new Maneuvers::FigureEight(maneuver, m_task, m_args->min_radius);
111  m_ltr = static_cast<Maneuvers::AbstractLoiter*>(fig);
112  break;
113  default:
114  Maneuvers::Circular* circ;
115  circ = new Maneuvers::Circular(maneuver, m_task, m_args->min_radius);
116  m_ltr = static_cast<Maneuvers::AbstractLoiter*>(circ);

126  void
127  onPathControlState(const IMC::PathControlState* pcs)
128  {
129  m_ltr->onPathControlState(pcs);
130 
131  if (m_ltr->isLoitering())
132  {
133  if (m_duration)
134  {
135  double now = Clock::get();
136 
137  if (m_end_time < 0)
138  {
139  m_task->debug("will now loiter for %d seconds", m_duration);
140  m_end_time = now + m_duration;
141  }
142  else if (now >= m_end_time)
143  {
144  m_task->signalCompletion();
145  }
146  else
147  {
148  m_task->signalProgress((uint16_t)Math::round(m_end_time - now));
149  }
150  }
151  }
152  else
153  {
154  if (m_duration > 0)
155  m_task->signalProgress(pcs->eta + m_duration);
156  else
157  m_task->signalProgress();
158  }
159  }
