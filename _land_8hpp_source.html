56  void
57  onStart(const IMC::Land* maneuver)
58  {
59  // Local variables
60  float dist;
61  double m_lat, m_lon;
62 
63  // Save touchdown desired coordinates
64  m_land = *maneuver;
65 
66  // Enable Path Control
67  m_task->setControl(IMC::CL_PATH);
68 
69  // Calculation of Glide Slope WP
70  m_lat = maneuver->lat;
71  m_lon = maneuver->lon;
72  dist = maneuver->glide_slope * maneuver->glide_slope_alt;
73  Coordinates::WGS84::displace(-std::cos(maneuver->bearing)*dist, -std::sin(maneuver->bearing)*dist, &m_lat, &m_lon);
74 
75  // Dispatch DesiredPath of Glide Slope WP
76  m_path.end_lat = m_lat;
77  m_path.end_lon = m_lon;
78  m_path.end_z = maneuver->glide_slope_alt;
79  m_path.end_z_units = maneuver->z_units;
80  m_path.speed = maneuver->speed;
81  m_path.speed_units = maneuver->speed_units;
82  m_path.lradius = 0;
83  m_path.flags = DesiredPath::FL_3DTRACK;
84  m_task->dispatch(m_path);
85 
86  m_task->debug("Sent DesiredPath of GlideSlope WP.");
87  }

91  void
92  onPathControlState(const IMC::PathControlState* pcs)
93  {
94  // Maneuver doesn't complete until vehicle reports vehicle medium as ground
95  if (pcs->flags & IMC::PathControlState::FL_NEAR)
96  {
97  if (!m_status)
98  {
99  sendTouchdown(m_land.lat, m_land.lon, m_land.z);
100  m_status = true;
101  }
102  }
103  else
104  m_task->signalProgress(pcs->eta);
105  }

110  onVehicleMedium(const IMC::VehicleMedium* msg)
111  {
112  m_ground = (msg->medium == IMC::VehicleMedium::VM_GROUND);
113 
114  // Maneuver is complete only vehicle medium is GROUND
115  if(m_ground && m_status)
