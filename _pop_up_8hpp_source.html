192  void
193  onVehicleMedium(const IMC::VehicleMedium* msg)
194  {
195  switch (m_pstate)
196  {
197  case ST_NEAR_SURFACE:
198  if (msg->medium == IMC::VehicleMedium::VM_WATER)
199  {
200  // Send report.

203  IMC::ReportControl rc;
204  rc.op = IMC::ReportControl::OP_REQUEST_REPORT;
205  rc.comm_interface = IMC::ReportControl::CI_SATELLITE
206  | IMC::ReportControl::CI_GSM;
207  rc.period = 0;
208  rc.sys_dst = "default";
209  m_task->dispatch(rc);
210  }
211 
212  if (mustKeep())

214  m_pstate = ST_SKEEP;
215  m_dur_timer.reset();

232  void
233  onGpsFix(const IMC::GpsFix* msg)
234  {
235  float dist;
236 
237  switch (m_pstate)
238  {
239  case ST_GET_FIX:
240  case ST_SKEEP:
241  if (mustWait())
242  return;
243 
244  if (!(msg->validity & IMC::GpsFix::GFV_VALID_POS))
245  break;
246 
247  double lat;
248  double lon;
249  Coordinates::toWGS84(m_state, lat, lon);
250 
251  dist = Coordinates::WGS84::distance(lat, lon, 0.0,
252  msg->lat, msg->lon, 0.0);
253 
254  if (msg->satellites >= m_args->min_sats &&
255  dist < m_args->min_distance)
256  {
257  goDown();
258  m_pstate = ST_GO_DOWN;
259  }
260  break;
261  default:
262  break;
263  }

270  {
271  m_state = *msg;
272 
273  switch (m_pstate)
274  {
275  case ST_GO_UP:

292  break;

300  void
301  onPathControlState(const IMC::PathControlState* pcs)
302  {
303  switch (m_pstate)
304  {
305  case ST_GO_TO:
306  if (pcs->flags & IMC::PathControlState::FL_NEAR)
307  {
308  goUp();
309  m_pstate = ST_GO_UP;
310  }
311  break;
312  case ST_GO_UP:
313  m_elevate->updatePathControl(pcs);
314 
315  // reached surface?
316  if (m_elevate->isDone())
317  {
318  m_task->setControl(IMC::CL_NONE);
319  m_pstate = ST_NEAR_SURFACE;
320  }
321 
322  break;
323  case ST_GO_DOWN:
324  m_elevate->updatePathControl(pcs);
325 
326  // all done?
327  if (m_elevate->isDone())
328  m_pstate = ST_DONE;
329 
330  break;
331  case ST_SKEEP:
332  m_skeep->updatePathControl(pcs);
333  break;
334  case ST_DONE:
335  m_task->signalCompletion();
336  break;
337  default:
338  break;
339  }
