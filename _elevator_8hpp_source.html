111  void
112  onPathControlState(const IMC::PathControlState* pcs)
113  {
114  m_elevate->updatePathControl(pcs);
115 
116  if (m_elevate->isDone())
117  {
118  m_task->signalCompletion();
119  return;
120  }
121 
122  float vertical_error = m_elevate->getVerticalError(m_depth, m_altitude);
123 
124  if (pcs->flags & IMC::PathControlState::FL_LOITERING)
125  {
126  if (m_args->vmonitor_speed > 0.0)
127  checkVerticalProgress();
128 
129  float time_left = vertical_error / (m_u_speed * std::sin(m_pitch));
130  m_task->signalProgress((unsigned)std::fabs(time_left));
131  }
132  else
133  {
134  float diagonal_length = vertical_error / std::sin(Plans::c_rated_pitch);
135  float loiter;
136 
137  if (m_u_speed)
138  loiter = diagonal_length / m_u_speed;
139  else
140  loiter = Plans::c_max_eta;
141 
142  m_task->signalProgress((uint16_t)Math::round(pcs->eta + loiter));
143  }
144  }

147  void
148  checkVerticalProgress(void)
149  {

151  // (multiply by minus direction for absolute value)
152  float vrate = - m_elevate->getElevatorDirection() * m_vz;
153 
154  if (m_vmon->isProgressSlow(vrate))
155  m_task->signalError(DTR("vertical progress is too slow"));
156  }
