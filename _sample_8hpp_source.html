85  void
86  onThrottle(const IMC::Throttle* throttle)
87  {
88  m_task->debug("Trottle: %f\tDone?: %d\n", throttle->value, m_done);
89 
90  if (throttle->value < 1 && (m_medium.medium == IMC::VehicleMedium::VM_GROUND) && !m_done)
91  {
92  if (m_maneuver.syringe0 == IMC::BOOL_TRUE)
93  setSyringeState(m_args->syringe0Id, m_args->openValue);
94 
95  if (m_maneuver.syringe1 == IMC::BOOL_TRUE)
96  setSyringeState(m_args->syringe1Id, m_args->openValue);
97 
98  if (m_maneuver.syringe2 == IMC::BOOL_TRUE)
99  setSyringeState(m_args->syringe2Id, m_args->openValue);
100 
101  m_done = true;
102  }
103 
104  if (timer.overflow())
105  {
106  if (m_done)
107  m_task->signalCompletion();
108  else
109  m_task->signalError(DTR("sample failed"));
110  }
111  else
112  m_task->debug("Sampling for %f/%f seconds...", timer.getElapsed(), timer.getTop());
113  }
