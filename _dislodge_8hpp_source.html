98  void
99  onEstimatedState(const IMC::EstimatedState* msg)
100  {
101  m_estate = *msg;
102 
103  if (!m_got_depth)
104  m_init_depth = msg->depth;
105 
106  switch (m_state)
107  {
108  case ST_START:
109  break;
110  case ST_BACK:
111  case ST_FRONT:
112  if (m_timer.overflow())
113  {
114  if (m_bursting)
115  {
116  ++m_bursts;
117  if (m_bursts >= m_args->bursts)
118  {
119  setState(ST_CHECK_RESULTS);
120  break;
121  }
122 
123  stopBurst();
124  m_bursting = false;
125  m_timer.setTop(m_args->interval_time);
126  }
127  else
128  {
129  startBurst(m_state);
130  m_bursting = true;
131  m_timer.setTop(m_args->burst_time);
132  }
133  }
134  break;
135  case ST_CHECK_RESULTS:
136  if (m_timer.overflow())
137  {
138  if (m_init_depth < m_args->safe_gap + 0.5)
139  {
140  if (msg->depth < m_args->depth_threshold)
141  {
142  m_task->signalCompletion();
143  setState(ST_DONE);
144  break;
145  }
146  }
147  else if (m_init_depth - msg->depth > m_args->safe_gap)
148  {
149  m_task->signalCompletion();
150  setState(ST_DONE);
151  break;
152  }
153 
154  failedAttempt();
155  }
156  break;
157  case ST_FAILED:
158  break;
159  case ST_DONE:
160  break;
161  }
162  }
