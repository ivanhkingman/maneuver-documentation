81  void
82  onStart(const IMC::YoYo* maneuver)
83  {
84  // Enable path, pitch and speed control
85  m_task->setControl(IMC::CL_PATH | IMC::CL_PITCH);
86 
87  IMC::DesiredPath path;
88  path.end_lat = maneuver->lat;
89  path.end_lon = maneuver->lon;
90 
91  path.speed = maneuver->speed;
92  path.speed_units = maneuver->speed_units;
93 
94  // disable depth control within path control
95  path.flags = IMC::DesiredPath::FL_NO_Z;
96  m_task->dispatch(path);
97 
98  m_zunits = static_cast<IMC::ZUnits>(maneuver->z_units);
99 
100  m_dispatched = false;
101 
102  if (maneuver->z_units != IMC::Z_ALTITUDE && maneuver->z_units != IMC::Z_DEPTH)
103  {
104  m_task->signalInvalidZ();
105  return;
106  }
107 
108  m_task->debug("altitude/depth :[%0.2f %0.2f] | %s: %0.2f %s | %s: %0.2f %s %d %s",
109  std::fabs(maneuver->z - maneuver->amplitude),
110  std::fabs(maneuver->z + maneuver->amplitude),
111  "pitch", Angles::degrees(maneuver->pitch),
112  "degrees", "speed", path.speed,
113  "in", (int)path.speed_units, "units");
114 
115  // basic init.
116  m_on_course = false;
117  m_course_recovered = false;
118 
119  // initialize yoyo motion controller
120  Memory::clear(m_yoyo);
121 
122  m_yoyo = new YoYoMotion(m_task, maneuver->pitch, maneuver->z,
123  (IMC::ZUnits)maneuver->z_units,
124  maneuver->amplitude, m_args->variation,
125  m_args->min_alt, m_args->max_depth);
126  }

130  void
131  onPathControlState(const IMC::PathControlState* pcs)
132  {
133  if (pcs->flags & IMC::PathControlState::FL_NEAR)
134  {
135  // done, stabilize pitch at the end
136  m_pitch.value = 0;
137  m_task->dispatch(m_pitch);
138  m_task->signalCompletion();
139  return;
140  }
141 
142  m_task->signalProgress(pcs->eta);
143 
144  // Check if off-track (stabilize pitch otherwise)
145  if (m_args->check_errors &&
146  (std::fabs(pcs->course_error) >= m_args->u_course ||
147  std::fabs(pcs->y) >= m_args->u_ctrack))
148  {
149  m_on_course = false;
150  m_course_recovered = false;
151  }
152  else if (!m_on_course)
153  {
154  m_on_course = true;
155  m_course_recovered = true;
156  }
157  }

188  void
189  update(double depth, double alt, double theta)
190  {
191  // Pitch value for control
192  double v;
193 
194  if (!m_on_course)
195  {
196  // Stabilize pitch at 0 degrees
197  v = m_yoyo->stabilize(0.0, theta);
198  }
199  else if (m_course_recovered)
200  {
201  // yo-yo again after course is recovered
202  v = m_yoyo->update(true, depth, alt, theta);
203  m_course_recovered = false;
204  }
205  else
206  {
207  v = m_yoyo->update(false, depth, alt, theta);
208  }
209 
210  // dont bother sending message if the value remains the same
211  // which will happen frequently
212  if (m_pitch.value == v && m_dispatched)
213  return;
214 
215  // Dispatch pitch message
216  m_pitch.value = v;
217  m_task->dispatch(m_pitch);
218 
219  m_dispatched = true;
220  }
