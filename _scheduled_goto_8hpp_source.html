82  void
83  onPathControlState(const IMC::PathControlState* pcs)
84  {
85  bool timeout = m_maneuver.arrival_time - Clock::getSinceEpoch() <= 0;
86  bool nearby = (pcs->flags & IMC::PathControlState::FL_NEAR) != 0;
87  bool delayed = m_speed >= m_args->max_speed;
88 
89  if (timeout && nearby)
90  {
91  m_task->signalCompletion();
92  return;
93  }
94 
95  if (timeout || delayed)
96  {
97  switch (m_maneuver.delayed)
98  {
99  case IMC::ScheduledGoto::DBEH_RESUME:
100  if (!nearby)
101  m_task->signalProgress(pcs->eta);
102  else
103  m_task->signalCompletion();
104  break;
105 
106  case IMC::ScheduledGoto::DBEH_SKIP:
107  m_task->signalCompletion();
108  break;
109 
110  case IMC::ScheduledGoto::DBEH_FAIL:
111  m_task->signalError(DTR("Unable to reach destination on scheduled time."));
112  break;
113  }
114  }
115  }

119  void
120  onEstimatedState(const IMC::EstimatedState* msg)
121  {
122  IMC::DesiredPath path;
123 
124  if (m_timer.overflow())
125  {
126  m_timer.reset();
127 
128  uint32_t control = IMC::CL_PATH | IMC::CL_SPEED;
129 
130  path.end_lat = m_maneuver.lat;
131  path.end_lon = m_maneuver.lon;
132  path.end_z = m_maneuver.z;
133  path.end_z_units = m_maneuver.z_units;
134 
135  switch (path.end_z_units)
136  {
137  case Z_DEPTH:
138  control |= IMC::CL_DEPTH;
139  break;
140  case Z_ALTITUDE:
141  control |= IMC::CL_ALTITUDE;
142  break;
143  default:
144  break;
145  }
146 
147  double prev_speed = m_speed;
148 
150  path.speed_units = IMC::SUNITS_METERS_PS;
151  m_speed = update(msg);
152  path.speed = m_speed;
153 
154  if (!m_dispatched && m_speed < m_args->min_speed)
155  {
156  // Start moving only when necessary...
157  m_task->setControl(0);
158  return;
159  }
160 
161  m_task->setControl(control);
162 
163  if (!m_dispatched || m_speed != prev_speed)
164  {
165  m_task->dispatch(path);
166  m_dispatched = true;
167  }
168  }
169  }

174  double
175  update(const IMC::EstimatedState* msg)
176  {
177  double lat;
178  double lon;
179  double dist;
180  double timeout;
181  double speed;
182 
184  Coordinates::toWGS84(*msg, lat, lon);
185 
187  dist = WGS84::distance(lat, lon, 0.0, m_maneuver.lat, m_maneuver.lon, 0.0);
188 
190  timeout = m_maneuver.arrival_time - Clock::getSinceEpoch();
191 
193  speed = dist / timeout;
194 
195  if (timeout < 0)
196  speed = m_args->max_speed;
197  else
198  speed = speed > m_args->max_speed ? m_args->max_speed : speed;
199 
200  m_task->debug("distance to target: %0.1f m(%0.1f s)", dist, timeout);
201  m_task->debug("new speed (max is %0.1f): %0.1f", m_args->max_speed, speed);
202 
203  return speed;
204  }
